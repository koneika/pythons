#include <iostream> // тут понятно

using namespace std; // чтоб не писать std::cout std::endl и тд.
const int N = 10; // размер матрицы, для массивов нужно число с const(что с перевода постоянное/не возможно меняющее число во время выполнения кода, наверно)

int main() // пишу int main или void main только надо убрать return 0;
{	
	int arr[N][N] = {0}; // = {число}, это мы говорим что в каждой переменной массива будет цифра 0	
	
	// в чём суть, это циклы 5*5, потому что в for(j) работает 5 раз, и каждая for(j) выполняет for(i) = 5 раз за каждый for(j)
	for(int j = 0; j < N; j++){
		for(int i = 0; i < N; i++){
			cout << arr[i][j] << " "; // 5 раз отображает в рядочек матрицу
		
		}
	
	cout << endl; // а потом делает отступ после пятого раза выводов 0
	}		



	return 0;
}
